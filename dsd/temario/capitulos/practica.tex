Implementar una barrera para
$N$ procesos usando un servidor central que no deje pasar a 
nadie hasta que el último haya llegado
Paso de Mensajes (Asíncrono/Canales).
port(id_cliente)
chan llegada[1..n]()
var esperando[1..n]: set of int
Servidor Barrera{
    contador=0
    while true do
    receive llegada(id_cliente)
    contador++
    esperando[contador] = id_cliente
    if contador == N then
        for i in 1..N do
            send llegada(esperando[i])
        contador = 0  
    
}

Servidor Parking{
    var plaza_disponible = MAX_PLAZAS;
    var esperando: queue of int;

    while true do
        receive peticion(id_cliente,operacion)
            switch(operacion)
                case ENTRAR:
                    if plaza_disponible > 0 && empty(esperando) then
                        plaza_disponible--;
                        send respuesta[id_cliente](1); // Aceptado
                    else
                        enqueue(esperando, id_cliente); // Espera
                case SALIR:
                    plaza_disponible++;
                    if not empty(esperando) then
                        proximo = dequeue(esperando);
                        plaza_disponible--;
                        send respuesta[proximo](1); // Aceptado al siguiente
    
}
